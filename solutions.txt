def add(a, b):
    return a + b

def loop(a, b):
    return a * b

def isit(a, b):
    if(a < b):
        return False
    else:
        return True

def operation(op, a, b):
    if op == '+':
        return a + b
    elif op == '-':
        return a - b
    elif op == '*':
        return a * b
    elif op == '/':
        return a / b
    else:
        return -1
